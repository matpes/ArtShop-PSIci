1. git init //u folderu gde zelimo da nam bude projekat
2. git config --global user.name "SanjaSamardzija"
3. git config --global user.email "sanja.z.samardzija@gmail.com"
.gitignore //fajl u kojem u svakom redu pisemo koje fajlove git treba da ignorise
4. git status //sta se desava u mom folderu
5. git log  //kako izgleda master i ostale grane
6. git add ime_fajla //dodajemo fajl u stage za komitovanje
7. git add .   //dodajemo sve fajlove u stage za komitovanje
8. git commit -m "porukica"  //komitujemo promene, ako ne stavimo -m onda nas prebaci u neki editor (Ctrl+C, :exit da se izadje kada se napise porukica)
9. git commit -a -m "porukica" //-||- + dodaje sve add-ovane u stage i komituje ih
10. git branch ime_grane    //napravi novu granu
11. git checkout ime_grane  //prelazimo na drugu granu
12. git merge ime_grane   //spaja tekucu i neku drugu granu
13. git pull ime_grane (origin)  //preuzmi sa servera sve
14. git push   //posalji na server sve moje komitovane fajlove
